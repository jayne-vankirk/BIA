# Computing the fantasy points for each player in the week 17 games

# List to hold total points of each player
players_fantasy_points = []
# Looping through each player
for name in player_names:
  total_points = 0
  # Looping through each category for fantasy points
  for category in fp_df['Category']:
    # if statements for all of the categories that have special conditions
    if category == 'Pass Yds (Greater than 300)':
      # gets the passing yards for that player
      pass_yds = week17_df.loc[week17_df['Name'] == name, 'Pass Yds']
      # checks if the yards are greater than 3000
      if pass_yds.iloc[0] >= 300:
        # if so it adds the points to the players fantasy score, otherwise it adds nothing
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    elif category == 'Rush Yds (Greater than 100)':
      rush_yds = week17_df.loc[week17_df['Name'] == name, 'Rush Yds']
      if rush_yds.iloc[0] >= 100:
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    elif category == 'Rcv Yds (Greater than 100)':
      rcv_yds = week17_df.loc[week17_df['Name'] == name, 'Rcv Yds']
      if rcv_yds.iloc[0] >= 100:
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    elif category == 'Ttl Pts Allowed (0)':
      allowed_points = week17_df.loc[week17_df['Name'] == name, 'Ttl Pts Allowed']
      if allowed_points.iloc[0] == 0:
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    elif category == 'Ttl Pts Allowed (1-6)':
      allowed_points = week17_df.loc[week17_df['Name'] == name, 'Ttl Pts Allowed']
      if allowed_points.iloc[0] >= 1 and allowed_points.iloc[0] <= 6:
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    elif category == 'Ttl Pts Allowed (7-13)':
      allowed_points = week17_df.loc[week17_df['Name'] == name, 'Ttl Pts Allowed']
      if allowed_points.iloc[0] >= 7 and allowed_points.iloc[0] <= 13:
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    elif category == 'Ttl Pts Allowed (14-20)':
      allowed_points = week17_df.loc[week17_df['Name'] == name, 'Ttl Pts Allowed']
      if allowed_points.iloc[0] >= 14 and allowed_points.iloc[0] <= 20:
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    elif category == 'Ttl Pts Allowed (21-27)':
      allowed_points = week17_df.loc[week17_df['Name'] == name, 'Ttl Pts Allowed']
      if allowed_points.iloc[0] >= 21 and allowed_points.iloc[0] <= 27:
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    elif category == 'Ttl Pts Allowed (28-34)':
      allowed_points = week17_df.loc[week17_df['Name'] == name, 'Ttl Pts Allowed']
      if allowed_points.iloc[0] >= 28 and allowed_points.iloc[0] <= 34:
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    elif category == 'Ttl Pts Allowed (35+)':
      allowed_points = week17_df.loc[week17_df['Name'] == name, 'Ttl Pts Allowed']
      if allowed_points.iloc[0] >= 35:
        total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0]
    # For all the other points, they are calculated by multiplying the occurences by the fantasy point values
    else:
      total_points += fp_df.loc[fp_df['Category'] == category, 'Points'].iloc[0] * week17_df.loc[week17_df['Name'] == name, category].iloc[0]
  # Points are added to the week17_df for that player once all the categories' points have been added
  week17_df.loc[week17_df['Name'] == name, 'Calculated Fantasy Points'] = total_points

Problem 1: Selecting the optimal lineup based on how the players performed during their actual games.

Objective: Maximize total fantasy points

Decision Variables:

  • Players (binary- if they were chosen yes/no)
  • Games (binary- if players chosen played in those games yes/no)

Constraints:

  • Exactly 9 players
  • Total salary of players cannot exceed 50,000
  • Players need to be from at least two different games
  • Positions needed: 1 QB, 2RBs, 3 WR, 1 TE, 1 DST, and 1 FLEX position (can be a RB, WR, or TE)

# Creating a model to determine what players we should choose based on their week 17 stats
model_p1 = pe.ConcreteModel()
# Binary decision variables for each player
model_p1.players = pe.Var(player_names, domain = pe.Binary)

# Extract unique games from the salary df
game_list = salaries_df['Game Info'].unique()

# Create binary decision variables for games
model_p1.game_selected = pe.Var(game_list, domain=pe.Binary)

# The objective is to maximize the fantasy points
# This loops through all players and based on if they were chosen or not, their calculated fantasy points
# are added to the total
model_p1.obj = pe.Objective(
    expr = sum(
        week17_df.loc[week17_df['Name'] == name, 'Calculated Fantasy Points'].values[0]
        * model_p1.players[name]
        for name in player_names),
    sense = pe.maximize)

Original Problem 2b: Creating a Custom Strategy for Choosing Players

I do not watch much football so I went on Google to find stats for each position. These are the sources I pulled inspiration from: https://www.reddit.com/r/fantasyfootball/comments/cb1kv8/predictive_fantasy_stats_simplified_a_quick_guide/ https://www.nbcsports.com/fantasy/football/news/2024-fantasy-football-d-st-draft-strategy

Objective: Maximize the following stats based on the player type:

  • QB: Cmp
  • RBs: Att
  • WRs and TEs: AY/A
  • DST: Pass Int

Decision Variables:

  • Players (binary- if they were chosen yes/no)
  • Games (binary- if players chosen played in those games yes/no)

Constraints:

  • Exactly 9 players
  • Total salary of players cannot exceed 50,000
  • Players need to be from at least two different games
  • Positions needed: 1 QB, 2RBs, 3 WR, 1 TE, 1 DST, and 1 FLEX position (can be a RB, WR, or TE)

# Creating a model based on maximizing various stats depending on position
model_custom = pe.ConcreteModel()
# Binary decision variables for each player
model_custom.players = pe.Var(player_names, domain = pe.Binary)

# Extract unique games from the salary df
game_list = salaries_df['Game Info'].unique()

# Create binary decision variables for games
model_custom.game_selected = pe.Var(game_list, domain=pe.Binary)

# Objective is to maximize stats based on the various player types
model_custom.obj = pe.Objective(
    expr = sum(
        # Cmp stat for selected QBs
        (pass_df.loc[pass_df['Name'] == player, 'Cmp'].values[0] * model_custom.players[player]
         if salaries_df[salaries_df['Name'] == player]['Position'].values[0] == 'QB' else 0) +
        # Att stat for selected RBs
        (rush_df.loc[rush_df['Name'] == player, 'Att'].values[0] * model_custom.players[player]

# Function to calculate average fantasy points across stats
def calculate_avg_fantasy_points(player_name):
    """Calculates the average fantasy points for a player using historical data."""
    # variable to hold total points for the player
    total_points = 0
    # variable to hold number of times the player shows up in a stat category: pass, rush, and rcv
    num_stats = 0

    # Get passing stats for the player and calculate fantasy points
    player_pass_df = pass_df[pass_df['Name'] == player_name]
    # goes through all the passing stats for that player
    for index, row in player_pass_df.iterrows():
        # calculates the points that should be added based on TDs, Yds, and if Yds was greater than 300 but subtracts any interceptions
        total_points += row['TD'] * 4 + row['Yds'] * 0.04 + (row['Yds'] > 300) * 3 - row['Int']
        num_stats += 1

    # Get rushing stats for the player and calculate fantasy points
    player_rush_df = rush_df[rush_df['Name'] == player_name]
    # goes through all the rushing stats for that player
    for index, row in player_rush_df.iterrows():
        # calculates the points that should be added based on Tds, Yds, and if Yds was greater than 100
        total_points += row['TD'] * 6 + row['Yds'] * 0.1 + (row['Yds'] > 100) * 3
        num_stats += 1

    # Get receiving stats for the player and calculate fantasy points
    player_rcv_df = rcv_df[rcv_df['Name'] == player_name]
    # goes through all the recieving stats for that player
    for index, row in player_rcv_df.iterrows():
        # calculates the points that should be added based on Tds, Yds, and if Yds was greater than 100, and Recs
        total_points += row['TD'] * 6 + row['Yds'] * 0.1 + (row['Yds'] > 100) * 3 + row['Rec']
        num_stats += 1

    # If it was able to find stats for that player it divides the total points by the number of stats it was able to find the player in
    if num_stats > 0:
        return total_points / num_stats
    else:
        return 0  # Return 0 if no historical data is found for the player
